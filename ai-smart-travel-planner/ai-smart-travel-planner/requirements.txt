import os
import time
from dotenv import load_dotenv
import streamlit as st
from google import genai
from google.genai.errors import APIError

# Load environment variables safely from your local .env file
load_dotenv()

# Pull from environment variables securely
GOOGLE_API_KEY = os.getenv('YOUR_GOOGLE_API_KEY')

# CLEAN CLIENT INITIALIZATION:
# Let the modern SDK handle default network timeouts natively. Bypasses the 
# 400 INVALID_ARGUMENT deadline restriction entirely.
client = genai.Client(api_key=GOOGLE_API_KEY)

# Initialize Session State variables for trip persistence
if 'travel_profile' not in st.session_state:
    st.session_state.travel_profile = {
        "Starting Point": "west Bengal, India",
        "destination": "Paris, France",
        "duration": "5 Days",
        "budget": "Moderate / Mid-Range",
        "travelers": "2 Adults",
        "interests": "History, Local Cuisine, Art Museums"
    }

if 'generated_itinerary' not in st.session_state:
    st.session_state.generated_itinerary = None

# Function to get Gemini Response with built-in retry-logic for Free Tiers
def get_travel_response(input_prompt):
    """
    Sends travel instructions to models/gemini-2.5-flash.
    Safely captures free-tier 429 and 503 exceptions to allow auto-retries.
    """
    try:
        response = client.models.generate_content(
            model="gemini-2.5-flash", 
            contents=[input_prompt]
        )
        return response.text
    
    except APIError as e:
        # Automated free-tier backoff logic for rate limits (429) AND temporary server overloads (503)
        if e.code in [429, 503]:  
            st.warning(f"Server reported error {e.code} (Busy/Rate Limit). Retrying automatically in 5 seconds...")
            time.sleep(5)
            return get_travel_response(input_prompt)
        return f"Gemini API Error: {str(e)}"
    except Exception as e:
        # Catch and auto-retry read/connection timeouts smoothly in the UI
        if "timed out" in str(e).lower() or "deadline" in str(e).lower():
            st.warning("The connection encountered a temporary network delay. Retrying the request automatically...")
            time.sleep(3)
            return get_travel_response(input_prompt)
        return f"Unexpected Error: {str(e)}"

# App Layout Configuration

        st.session_state.travel_profile = {

            "duration": days,
